import { firebase, db } from "../firebase";

const [currentLoggedInUser, setCurrentLoggenInUser] = useState(null);

const getHighscore = () => {
    const user = firebase.auth().currentUser;
    const unsubscribe = db
      .collection("users")
      .where("owner_uid", "==", user.uid)
      .limit(1)
      .onSnapshot((snapshot) =>
        snapshot.docs.map((doc) => {
          setCurrentLoggenInUser({
            highscore: doc.data().highscore,
          });
        })
      );
    return unsubscribe;
  };

  useEffect(() => {
    getHighscore();
  }, []);
